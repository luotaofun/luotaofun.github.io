# _*_ coding : utf-8 _*_
# @Time : 2023/12/17 12:56

"""
1.python中系统自动辨别数据类型,并不需要显示的去声明
2.标识符由字母/数字/下划线构成且大小写敏感
3.python中都是字符串才可加法拼接
"""
name_list = ["罗涛","周超"] # 列表
age_tuple = (18,19) # 元祖
person = {"name" : "罗涛","age" : 18} # 字典

# 强制类型转换.0或数据为空则为False
a = True
print(type(a)) # 判断数据类型
a1 = str(a) # 布尔转字符串
print(type(a1) )
print(a1)

b = " " # 空格转布尔TRUE
# b = "" # 空字符串转布尔FALSE
print(type(b))
b1 = bool(b)
print(type(b1))
print(b1)

print(1/2) # 商
print(1//2) # 取整
print(1 ** 2) # 幂

# 多变量赋值
a,b,c = 1,2,3

# 短路与\短路或
True and print(1) # 1
False and print(2) # 不执行

True or print(3) # 3
False or print(4) # 4

# 格式化输出
name = "罗涛"
age = 18
print("我的名字是%s,我的年龄是%d" %(name,age))

# 输入
print(input("请输入:"))

# 流程控制
if int(input("请输入你的年龄:")) > 18:
    print(True)
else:
    print("输入错误!")
    
# 分支判断
score = int(input("请输入成绩:"))
if score >=90:
    print("优秀")
elif score >=60:
    print("合格")
else:
    print("不合格")
    
# 遍历    
name_list = ["罗涛","周超"]
for i in range(len(name_list)): # 遍历下标
    print(i) # 0 1
    
print(len(name_list)) # 列表长度 2

# 字符处理
s = "aa,bb "
print(s.find("b")) # 第一次出现下标 3
print(s.count("b")) # 出现次数 2
print(s.replace("b","c")) # aa,cc 
print(s.split(",")) # 切割为列表 ['aa', 'bb ']
print(s.strip(",")) # aa,bb移除字符串头尾指定的字符（默认为空格或换行符）
print(s.join("cc")) # caa,bb c 将字符串cc中的每个成员以字符'aa,bb '分隔开再拼接成一个字符串

"""
