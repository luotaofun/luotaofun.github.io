项目概述

这是一个基于Spring Boot和MyBatis的Java Web应用程序，名为”demo-luotaofun”。该项目使用Maven作为构建工具，采用了MVC架构模式，主要用于展示Spring Boot与MyBatis的集成使用。前端使用纯Vue.js 3构建，实现了一个现代化的区域管理系统界面。

/**
 * @Classname DataSourceConfiguration
 * @Description TODO
 * @Version 1.0.0
 * @Date 2025/3/16 14:53
 * @Author LuoTao
 */
@Configuration // 告诉容器需要来这里检索Bean
@MapperScan("com.luotao.demo.dao") // mybatis_mapper的扫描路径
public class DataSourceConfiguration{
    // 读取配置文件里面的属性值
    @Value("${jdbc.driver}")
    private String jdbcDriver;
    @Value("${jdbc.url}")
    private String jdbcUrl;
    @Value("${jdbc.username}")
    private String jdbcUsername;
    @Value("${jdbc.password}")
    private String jdbcPassword;

    @Bean(name="dataSource")
    public ComboPooledDataSource createDataSource() throws PropertyVetoException {
        ComboPooledDataSource dataSource = new ComboPooledDataSource();
        dataSource.setDriverClass(jdbcDriver);
        dataSource.setJdbcUrl(jdbcUrl);
        dataSource.setUser(jdbcUsername);
        dataSource.setPassword(jdbcPassword);
        // 关闭连接后不自动commit
        dataSource.setAutoCommitOnClose(false);
        return dataSource;
    }
}

<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.luotao.demo.dao.AreaDao">
    <insert id="insertArea" useGeneratedKeys="true" keyProperty="areaId" keyColumn="area_id"
            parameterType="com.luotao.demo.mybatis.entity.Area">
        insert into tb_area(area_name, priority, create_time, last_edit_time)
        values (#{areaName}, #{priority}, #{createTime}, #{lastEditTime})
    </insert>

    <delete id="deleteArea">
        delete
        from tb_area
        where area_id = #{areaId}
    </delete>

    <update id="updateArea" parameterType="com.luotao.demo.mybatis.entity.Area">
        update tb_area
        <set>
            <if test="areaName !=null">area_name=#{areaName}</if>
            <if test="priority !=null">priority=#{priority}</if>
            <if test="lastEditTime !=null">last_edit_time=#{lastEditTime}</if>
        </set>
        where area_id=#{areaId}
    </update>

    <select id="queryAreaById" resultType="com.luotao.demo.mybatis.entity.Area">
        select area_id, area_name, priority, create_time, last_edit_time
        from tb_area
        where area_id = #{areaId}
    </select>
    <select id="queryArea" resultType="com.luotao.demo.mybatis.entity.Area">
        select area_id, area_name, priority, create_time, last_edit_time
        from tb_area
        order by priority
                desc
    </select>
</mapper>

service的接口和dao的接口有什么区别吗？

DAO（Data Access Object）接口：
DAO 实现类直接与数据库交互，执行具体的 SQL 操作。
是数据访问层的核心组件，封装了对数据库的操作逻辑。
通常只关注数据的存储和检索，不涉及业务逻辑。
Service 接口：

整合复杂业务逻辑，将dao的增删改查整合

可能会调用一个或多个 DAO 接口来完成复杂的业务操作。
提供更高层次的功能抽象，屏蔽底层的数据访问细节。

Service 接口的方法更注重业务语义，例如 addArea 表示添加区域，modifyArea 表示修改区域。通常由 Service 层管理事务。
方法可能包含额外的校验逻辑或调用多个 DAO 方法来完成复杂的业务需求。

service层的实现类
package com.luotao.demo.service.impl;

import com.luotao.demo.dao.AreaDao;
import com.luotao.demo.mybatis.entity.Area;
import com.luotao.demo.service.AreaService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.Date;
import java.util.List;

/**
 * @Classname AreaServiceImpl
 * @Description TODO
 * @Version 1.0.0
 * @Date 2025/3/16 19:00
 * @Author LuoTao
 */
@Service
@Transactional(rollbackFor = RuntimeException.class)
public class AreaServiceImpl implements AreaService {
    @Autowired
    private AreaDao areaDao;

    @Override
    public boolean addArea(Area area) {
        if(area.getAreaName() != null && !"".equals(area.getAreaName())){
            area.setCreateTime(new Date());
            area.setLastEditTime(new Date());
            try{
                int effectedNum = areaDao.insertArea(area);
                if(effectedNum >0){
                    return true;
                }else {
                    throw new RuntimeException("插入区域信息失败!");
                }
            }catch (Exception e){
                throw new RuntimeException("插入区域信息失败:" + e.toString());
            }
        }else {
            throw new RuntimeException("区域信息不能为空！");
        }

    }

    @Override
    public boolean deleteArea(int areadId) {
        if(areadId >0){
            try{
                // 删除区域信息
                int effectedNum = areaDao.deleteArea(areadId);
                if (effectedNum >0){
                    return true;
                }else {
                    throw new RuntimeException("删除区域信息失败！");
                }
            }catch (Exception e){
                throw new RuntimeException("删除区域信息失败:" + e.toString())
            }
        }else{
            throw new RuntimeException("区域Id不能为空！");
        }
    }

    @Override
    public boolean modifyArea(Area area) {
        // id非空才能修改
        if(area.getAreaId() != null && !"".equals(area.getAreaId())){
            area.setLastEditTime(new Date());
            try{
                int effectedNum = areaDao.updateArea(area);
                if(effectedNum >0){
                    return true;
                }else {
                    throw new RuntimeException("更新区域信息失败!");
                }
            }catch (Exception e){
                throw new RuntimeException("更新信息失败:" + e.toString());
            }
        }else {
            throw new RuntimeException("区域信息不能为空！");
        }
    }

    @Override
    public Area getAreaById(int areaId) {
        return areaDao.queryAreaById(areaId);
    }

    @Override
    public List<Area> getAreaList() {
        return areaDao.queryArea();
    }
}

创建了AreaController，提供RESTful API：

package com.luotao.demo.web;

import com.luotao.demo.mybatis.entity.Area;
import com.luotao.demo.service.AreaService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * @Classname AreaController
 * @Description TODO
 * @Version 1.0.0
 * @Date 2025/3/16 19:53
 * @Author LuoTao
 */
@RestController
@RequestMapping("/superadmin")
public class AreaController {
    @Autowired
    private AreaService areaService;
    @RequestMapping(value = "/addarea", method = RequestMethod.POST)
    public Map<String, Object> addArea(@RequestBody Area area) {
        HashMap<String, Object> modeMap = new HashMap<>();
        modeMap.put("success", areaService.addArea(area));
        return modeMap;
    }

    @RequestMapping(value = "/removearea", method = RequestMethod.GET)
    public Map<String, Object> removeArea(Integer areaId) {
        HashMap<String, Object> modeMap = new HashMap<>();
        modeMap.put("success", areaService.deleteArea(areaId));
        return modeMap;
    }
    @RequestMapping(value = "/modifyarea", method = RequestMethod.POST)
    public Map<String, Object> modifyArea(@RequestBody Area area) {
        HashMap<String, Object> modeMap = new HashMap<>();
        modeMap.put("success", areaService.modifyArea(area));
        return modeMap;
    }

    @RequestMapping(value = "/listarea", method = RequestMethod.GET)
    public Map<String, Object> listArea() {
        HashMap<String, Object> modeMap = new HashMap<>();
        List<Area> areaList = areaService.getAreaList();
        modeMap.put("areaList", areaList);
        return modeMap;
    }
    @RequestMapping(value = "/getareabyid", method = RequestMethod.GET)
    public Map<String, Object> getAreaById(Integer areaId) {
        HashMap<String, Object> modeMap = new HashMap<>();
        Area area = areaService.getAreaById(areaId);
        modeMap.put("area", area);
        return modeMap;

// 使用Vue 3的Composition API
const { createApp, ref, onMounted } = Vue;

const app = createApp({
    setup() {
        // 区域列表
        const areas = ref([]);

        // 当前操作的区域对象
        const currentArea = ref({
            areaId: null,
            areaName: '',
            priority: 0
        });

        // 模态框显示状态
        const showModal = ref(false);
        const showDeleteModal = ref(false);

        // 是否处于编辑模式
        const isEditing = ref(false);

        // 页面加载时执行
        onMounted(() => {
            // 加载区域列表
            loadAreas();
        });

        // 加载区域列表
        const loadAreas = () => {
            axios.get('/superadmin/listarea')
                .then(response => {
                    if (response.data && response.data.areaList) {
                        areas.value = response.data.areaList;
                    } else {
                        areas.value = [];
                        console.error('获取区域列表失败:', response);
                    }
                })
                .catch(error => {
                    console.error('获取区域列表出错:', error);
                    alert('获取区域列表失败，请查看控制台了解详情。');
                });
        };

        // 显示添加区域模态框
        const showAddModal = () => {
            isEditing.value = false;
            currentArea.value = {
                areaId: null,
                areaName: '',
                priority: 0
            };
            showModal.value = true;
        };

        // 关闭模态框
        const closeModal = () => {
            showModal.value = false;
        };

        // 关闭删除确认模态框
        const closeDeleteModal = () => {
            showDeleteModal.value = false;
        };

        // 显示编辑区域模态框
        const editArea = (area) => {
            isEditing.value = true;
            // 创建一个新对象，避免直接修改原始数据
            currentArea.value = {
                areaId: area.areaId,
                areaName: area.areaName,
                priority: area.priority
            };
            showModal.value = true;
        };

        // 保存区域（添加或更新）
        const saveArea = () => {
            const url = isEditing.value ? '/superadmin/modifyarea' : '/superadmin/addarea';

            axios.post(url, currentArea.value)
                .then(response => {
                    if (response.data && response.data.success) {
                        // 关闭模态框
                        showModal.value = false;
                        // 重新加载区域列表
                        loadAreas();
                        // 显示成功消息
                        alert(isEditing.value ? '区域更新成功！' : '区域添加成功！');
                    } else {
                        console.error('保存区域失败:', response);
                        alert('操作失败，请查看控制台了解详情。');
                    }
                })
                .catch(error => {
                    console.error('保存区域出错:', error);
                    alert('操作失败，请查看控制台了解详情。');
                });
        };

        // 显示删除确认模态框
        const confirmDelete = (area) => {
            currentArea.value = area;
            showDeleteModal.value = true;
        };

        // 删除区域
        const deleteArea = () => {
            axios.get('/superadmin/removearea', {
                params: {
                    areaId: currentArea.value.areaId
                }
            })
                .then(response => {
                    if (response.data && response.data.success) {
                        // 关闭模态框
                        showDeleteModal.value = false;
                        // 重新加载区域列表
                        loadAreas();
                        // 显示成功消息
                        alert('区域删除成功！');
                    } else {
                        console.error('删除区域失败:', response);
                        alert('删除失败，请查看控制台了解详情。');
                    }
                })
                .catch(error => {
                    console.error('删除区域出错:', error);
                    alert('删除失败，请查看控制台了解详情。');
                });
        };

        // 格式化日期
        const formatDate = (dateString) => {
            if (!dateString) return '未设置';

            const date = new Date(dateString);
            if (isNaN(date.getTime())) return dateString;

            return date.toLocaleString('zh-CN', {
                year: 'numeric',
                month: '2-digit',
                day: '2-digit',
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });
        };

        // 返回所有需要在模板中使用的变量和方法
        return {
            areas,
            currentArea,
            isEditing,
            showModal,
            showDeleteModal,
            loadAreas,
            showAddModal,
            closeModal,
            closeDeleteModal,
            editArea,
            saveArea,
            confirmDelete,
            deleteArea,
            formatDate
        };
    }
});

// 挂载Vue应用
app.mount('#app');

@Autowired与@Resource的区别

@Autowired 注解的作用就是让 Spring 容器自动为 areaDao 成员变量注入一个实现了 AreaDao 接口的实例对象。Spring 会根据配置（例如 MyBatis 的配置）动态生成AreaDao 实现类的代理对象。这个代理对象会被注入到 areaDao 中

@Autowired根据type注入，@Resource基于名称，找不到时基于类型注入，本质上均实现了注入效果。
一般来说，注入controller的service虽然一般来说我们都是注入一个接口，但是该接口有实现类，并且使用@Service进行关联，所以注入类型应该也可以视为一个类，但是mybatis仅需提供Dao接口，也就是说，注入service的dao只是一个接口，而没有实现类，虽然mybatis能够通过Dao接口和xml文件实现与数据库的操作，但是@Autowired并没有这个识别功能，可能它就认为你类型不匹配，无法使用通过类型注入的方法
