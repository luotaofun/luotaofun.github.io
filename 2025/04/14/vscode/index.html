基础配置

{

    // ============== 外观与主题配置 ==============
    "workbench.colorTheme": "Idea intellij light theme",  // JetBrains风格亮色主题
    "workbench.preferredDarkColorTheme": "Idea intellij light theme",  // 统一亮色主题
    "workbench.iconTheme": "jetbrainsicons",              // JetBrains风格图标（需安装插件）
    "editor.minimap.enabled": false,                     // 禁用右侧代码缩略图

    // ============== 编辑器基础配置 ==============
    "editor.fontFamily": "Consolas",                     // Windows经典等宽字体
    "editor.fontSize": 14,                               // 适中字号
    "editor.fontLigatures": true,                        // 启用编程连字（如=>显示为箭头）
    "editor.mouseWheelZoom": true,                       // 支持Ctrl+滚轮缩放编辑器
    "window.zoomLevel": 2,                               // 整体界面缩放（根据显示器DPI调整）
    
    // ============== 光标与动画配置 ==============
    "editor.cursorSmoothCaretAnimation": "on",           // 光标平滑移动动画
    "editor.cursorBlinking": "smooth",      // 光标平滑闪烁效果
                 
    // ============== 工作区行为配置 ==============
    "window.restoreWindows": "none",                     // 启动时不恢复上次会话
    "workbench.startupEditor": "newUntitledFile",        // 启动时创建新文件

    // ============== 终端配置 ==============
     // "terminal.integrated.defaultProfile.windows": "Command Prompt", 
        // Git Bash
        "terminal.integrated.profiles.windows": {
            "Git-Bash": {
                "path": "D:\\software\\Git\\bin\\bash.exe", 
                "args": ["--login"],
                "icon": "terminal-bash"
            }
        },
        "terminal.integrated.defaultProfile.windows": "Git Bash", // 设为默认终端
    
        // 环境变量配置（增强Git Bash兼容性）
        "terminal.integrated.env.windows": {
            "MSYS_NO_PATHCONV": "1"
        },
    

    // ============== 语言与框架支持 ==============
    // Vue开发支持（需安装Volar插件）
    "[vue]": {
        "editor.defaultFormatter": "Vue.volar"           // Vue官方格式化工具
    },
    
    // HTML格式化配置（需安装Prettier插件）
    "[html]": {
        "editor.defaultFormatter": "esbenp.prettier-vscode"
    },

    // ============== 路径与自动补全配置 ==============
    // @符号路径提示（需安装Path Autocomplete插件）
    "path-autocomplete.pathMappings": {
        "@": "${folder}/src"  // 适用于Vue/React等基于src目录的项目结构
    },

    // ============== 其他工具配置 ==============

    // Emmet模板配置
    "emmet.variables": {
        "lang": "zh-CN",                                 // 中文文档
        "charset": "UTF-8"                               // 统一编码格式
    },
    
    // console-ninja插件
    "console-ninja.featureSet": "Community" ,             

    // Live Server配置（需安装Live Server插件）
    "liveServer.settings.CustomBrowser": "microsoft-edge", // 指定Edge作为调试浏览器
    
    // GitHub Copilot配置
    "github.copilot.enable": {
        "*": false,                                      // 全局禁用AI代码补全
        "plaintext": false,
        "markdown": false,
        "scminput": false,
        "html": false
    },



}
